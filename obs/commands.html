Система

Маршрутизация сообщений

Куда попадает каждое сообщение из чата

flowchart TD
    IN["Сообщение в чате GG"] --> R{"Первая команда?"}
    R -->|!команды| HELP["Показать список команд"]
    R -->|иначе| P["Модуль Princess"]
    P --> PH{"Обработал?"}
    PH -->|да| OUT["Ответ в чат"]
    PH -->|нет| RO["Модуль Roulette"]
    RO --> RH{"Обработал?"}
    RH -->|да| OUT
    RH -->|нет| S["Модуль Song Request"]
    S --> SH{"Обработал?"}
    SH -->|да| OUT
    SH -->|нет| IGN["Игнор"]
    class IN,P,RO,S action
    class R,PH,RH,SH cond
    class HELP,OUT ok
    class IGN muted
Система

Обработка каждого сообщения

Что происходит до разбора конкретной команды

flowchart TD
    M["Любое сообщение"] --> SAVE["Запомнить ник"]
    SAVE --> J{"В тюрьме?"}
    J -->|да| SR{"Команда !срок?"}
    SR -->|да| TIME["Показать время до выхода"]
    SR -->|нет| BLOCK["Остальные команды заблокированы"]
    J -->|нет| PTS["+принцессы за сообщение"]
    PTS --> C{"Начинается с ! ?"}
    C -->|нет| PASS["Передать в Song Request"]
    C -->|да| ROUTE["Princess / Админ / неизвестная"]
    class M,SAVE,PASS,ROUTE action
    class J,SR,C cond
    class TIME ok
    class BLOCK fail
    class PTS eco
Параллельно: зрители в чате (не в тюрьме) получают пассивный доход каждую минуту

!дейлик — ежедневный бонус

Раз в сутки, размер бонуса зависит от номера дня в месяце

flowchart TD
    START["!дейлик"] --> CHK{"Уже получал сегодня?"}
    CHK -->|да| NO["Отказ"]
    CHK -->|нет| YES["Начислить бонус"]
    class START action
    class CHK cond
    class YES ok
    class NO fail

!дайс — бросок кубика

Платный бросок d20, перезарядка 5 минут

flowchart TD
    START["!дайс"] --> CHK{"КД прошёл и хватает баллов?"}
    CHK -->|нет| NO["Отказ"]
    CHK -->|да| PAY["Списать стоимость броска"]
    PAY --> ROLL["Бросок d20"]
    ROLL -->|1| CF["Крит. провал −50"]
    ROLL -->|2–19| OK["Показать результат"]
    ROLL -->|20| CS["Крит. успех +200"]
    class START,PAY,ROLL action
    class CHK cond
    class OK,CS ok
    class NO,CF fail

!кража — кража у зрителя

Только среда и пятница, минимум 5 зрителей в чате, КД 10 минут

flowchart TD
    START["!кража"] --> D1{"Среда или пятница?"}
    D1 -->|нет| F0["Отказ"]
    D1 -->|да| D2{"≥ 5 зрителей?"}
    D2 -->|нет| F1["Отказ"]
    D2 -->|да| D3{"КД 10 мин?"}
    D3 -->|нет| F2["Отказ"]
    D3 -->|да| ROLL["Бросок на шанс кражи"]
    ROLL -->|провал| F3["Ничего не украл"]
    ROLL -->|успех| VICT["Случайный зритель — жертва"]
    VICT --> D4{"У жертвы ≥ 18к?"}
    D4 -->|нет| F4["Отказ с шуткой"]
    D4 -->|да| LOOT["Перевести принцессы"]
    LOOT --> D5{"Попал в тюрьму?"}
    D5 -->|да| JAIL["Блок на 30 минут"]
    D5 -->|нет| WIN["Успех"]
    class START,ROLL,VICT action
    class D1,D2,D3,D4,D5 cond
    class WIN ok
    class F0,F1,F2,F3,F4,JAIL fail
    class LOOT eco
Roulette

!рулетка — общий раунд на чат

Одна ставка на игрока за раунд; выплаты из казны модуля

flowchart TD
    R0["!рулетка правила"] --> RULES["Краткая справка"]
    R1["!рулетка сумма ..."] --> RS{"Стол открыт / cooldown / казна?"}
    RS -->|нет| RF["Отказ"]
    RS -->|да| RD{"Уже ставил в раунде?"}
    RD -->|да| RF2["Отказ без списания"]
    RD -->|нет| RB{"Хватает баллов?"}
    RB -->|нет| RF3["Недостаточно баллов"]
    RB -->|да| BET["Списать баллы → казна → запись ставки"]
    BET --> OPEN{"Первый в авто-режиме?"}
    OPEN -->|да| TMR["Открыть стол + таймер сбора"]
    OPEN -->|нет| WAIT["Ждать спина"]
    TMR --> SPIN["Спин 0–36"]
    WAIT --> SPIN
    SPIN --> WIN{"Есть победители?"}
    WIN -->|нет| LOSE["Ставки остаются в казне"]
    WIN -->|да| PAY["Выплата из казны"]
    PAY --> CUT{"Хватает казны?"}
    CUT -->|нет| PART["Пропорционально урезать"]
    CUT -->|да| FULL["Полная выплата"]
    class R0,R1,BET,TMR,SPIN action
    class RS,RD,RB,OPEN,WIN,CUT cond
    class RULES,FULL ok
    class RF,RF2,RF3,LOSE,PART fail
    class BET,PAY eco
Скрытая справка: !рулетка правила. Админ: !рулетка_банк, !рулетка_пополнить, !рулетка_сброс.

Song Request

!заказ — заказ трека

Также: !зм и !sr + ссылка на YouTube

flowchart TD
    START["!заказ + URL"] --> E1{"Заказы включены?"}
    E1 -->|нет| F0["Отказ"]
    E1 -->|да| E2{"КД пользователя?"}
    E2 -->|нет| F1["Подожди"]
    E2 -->|да| E3{"Очередь не полна?"}
    E3 -->|нет| F2["Очередь заполнена"]
    E3 -->|да| E4{"Валидный YouTube?"}
    E4 -->|нет| F3["Неверная ссылка"]
    E4 -->|да| E5{"Хватает принцесс?"}
    E5 -->|нет| F4["Недостаточно"]
    E5 -->|да| PAY["Списать стоимость → очередь"]
    PAY --> E6{"Ничего не играет?"}
    E6 -->|да| PLAY["Сразу воспроизвести"]
    E6 -->|нет| WAIT["Ждать в очереди"]
    PAY -.->|ошибка плеера| REFUND["Возврат принцесс"]
    class START action
    class E1,E2,E3,E4,E5,E6 cond
    class PLAY,WAIT ok
    class F0,F1,F2,F3,F4 fail
    class PAY eco
    class REFUND fail
